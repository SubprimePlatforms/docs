  <div className="deepidv-feature-card">

      <div className="deepidv-feature-card-icon">
        <Icon icon={icon} size={20} />
      </div>
    )}
    <div className="deepidv-feature-card-content">
      <h3 className="deepidv-feature-card-title">
        {title}
        {badge && <span style={{ fontSize: "0.7rem", fontWeight: 500, color: "#1E7FE0", marginLeft: "0.5rem", background: "#EBF4FE", padding: "2px 8px", borderRadius: "9999px" }}>{badge}</span>}
      </h3>
      {description && (
        <p className="deepidv-feature-card-desc">{description}</p>
      )}
    </div>
  </div>
